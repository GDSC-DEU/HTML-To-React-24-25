React에서 주로 사용하는 ES6 문법

화살표 함수 (Arrow Function)

화살표 함수는 function 키워드 대신 => 기호를 사용하여 함수를 선언하는 방식


Before => 기존에 사용하는 방식


            hello = function() {
                return "Hello World!";
            }
            document.getElementById("before").innerHTML = hello();


After => ES6 문법을 사용한 방식


            hello = () => {
                return "Hello World!";
            }
            document.getElementById("after").innerHTML = hello();


만약 parameter 있는 경우


            hello = (val) => "Hello " + val;
    
            document.getElementById("parameter").innerHTML = hello("User!");



구조 분해 할당 (Destructuring Assignment)

Before => 기존에 사용하는 방식


            let vehicles = ['Grandeur', 'Santa Fe', 'Elec City'];

            const car = vehicles[0];
            const suv = vehicles[1];
            const bus = vehicles[2];

            document.getElementById("array").innerHTML = car + ", " + suv + ", " + bus;


After => ES6 문법을 사용한 방식


            [car, suv, bus] = vehicles;

            document.getElementById("destructuring").innerHTML = car + ", " + suv + ", " + bus;


원하는 단어만 추출 가능!


            [car, , bus] = vehicles;

            document.getElementById("skip").innerHTML = car + ", " + bus;


Example


            let vehicleOne = {
                brand: 'Hyundai',
                model: 'Grandeur',
                type: 'Sedan',
                year: 2022,
                color: 'Black'
            }
        
            myVehicle(vehicleOne);
        
            function myVehicle({model, color, brand}) {
                const introduce = 'My car is ' + model + ' color is ' + color + ' and brand is ' + brand;
        
                document.getElementById("object").innerHTML = introduce;
            }



펼침 연산자 (Spread Operator)

Example1


            let numbersOne = [1, 2, 3];
            let numbersTwo = [4, 5, 6];
            let numbersCombined = [...numbersOne, ...numbersTwo];
        
            document.getElementById("spread").innerHTML = numbersCombined;


Example2


            let numbers = [1, 2, 3, 4, 5, 6];

            let [one, two, ...rest] = numbers;
        
            document.getElementById("One").innerHTML = one;
            document.getElementById("Two").innerHTML = two;
            document.getElementById("Rest").innerHTML = rest;


Example3


            let myVehicle = {
                brand: 'Hyundai',
                model: 'Grandeur',
                color: 'Black'
            }
        
            let updateMyVehicle = {
                type: 'Sedan',
                year: 2022,
                color: 'White'
            }
        
            let myUpdatedVehicle = {...myVehicle, ...updateMyVehicle};
        
            document.getElementById("objectSpread").innerHTML = 'My car is ' + myUpdatedVehicle.model + ' color is ' + myUpdatedVehicle.color + ' and brand is ' + myUpdatedVehicle.brand;
